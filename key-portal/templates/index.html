📖 Claude Code 使用教程

1

执行配置命令

申请成功后，复制显示的"Claude Code 一键配置命令"，粘贴到终端执行： 

node << 'CLAUDE_EOF'
const fs=require('fs'),path=require('path'),os=require('os');
const dir=path.join(os.homedir(),'.claude');
const sf=path.join(dir,'settings.json');
if(!fs.existsSync(dir))fs.mkdirSync(dir,{recursive:true});
const s=fs.existsSync(sf)?JSON.parse(fs.readFileSync(sf,'utf8')):{};
s.apiKeyHelper='echo YOUR_API_KEY';
if(!s.env)s.env={};
s.env.ANTHROPIC_BASE_URL='http://当前页面IP:8317';
fs.writeFileSync(sf,JSON.stringify(s,null,2));
console.log('Done!');
CLAUDE_EOF

💡 说明：这条命令会合并更新 ~/.claude/settings.json，只修改 API Key 和服务器地址，保留你已有的 MCP、permissions 等配置

2

验证配置

运行 claude 命令，Claude Code 会自动读取配置的 API Key 和 Base URL 

$ claude
✓ Configuration loaded from ~/.claude/settings.json
# 看到这个提示说明配置成功
3

开始使用

现在可以正常使用 Claude Code 了！以下是一些常用命令示例：

基础对话：

claude "帮我写一个Python快速排序函数"

代码项目：

cd your-project
claude "分析这个项目的代码结构"

交互模式：

claude  # 直接进入交互式对话

💡 重要提示

  • 配置文件保存在 ~/.claude/settings.json，重启终端后依然有效
  • 如需更换 API Key，只需重新执行配置命令即可覆盖
  • 多台电脑使用：每台电脑都需要执行一次配置命令

手动配置方式（可选）

如果需要手动配置，可以创建 ~/.claude/settings.json 文件： 

{
  "apiKeyHelper": "echo YOUR_API_KEY",
  "env": {
    "ANTHROPIC_BASE_URL": "http://当前页面IP:8317"
  }
}

🗑 安全移除代理配置

如需恢复为官方 Claude Code（仅移除代理相关配置，保留 MCP、permissions 等设置）： 

node << 'REMOVE_EOF'
const fs=require('fs'),path=require('path'),os=require('os');
// Remove Claude proxy config
const sf=path.join(os.homedir(),'.claude','settings.json');
if(fs.existsSync(sf)){
  const s=JSON.parse(fs.readFileSync(sf,'utf8'));
  delete s.apiKeyHelper;
  if(s.env){Object.keys(s.env).forEach(k=>{if(k.startsWith('ANTHROPIC'))delete s.env[k]});if(Object.keys(s.env).length===0)delete s.env}
  fs.writeFileSync(sf,JSON.stringify(s,null,2));
  console.log('Claude: proxy config removed (MCP/permissions preserved)');
}else{console.log('Claude: no settings.json, skipped')}
// Remove Codex proxy config
const cf=path.join(os.homedir(),'.codex','config.toml');
if(fs.existsSync(cf)){
  let c=fs.readFileSync(cf,'utf8');
  c=c.replace(/^openai_base_url\s*=.*\n?/m,'');
  c=c.replace(/^model_provider\s*=\s*"local"\n?/m,'');
  c=c.replace(/\[model_providers\.local\][\s\S]*?(?=\n\[|$)/m,'');
  fs.writeFileSync(cf,c);
  console.log('Codex: local proxy provider removed from '+cf);
}else{console.log('Codex: no config.toml, skipped')}
// Remove OPENAI_API_KEY from shell rc
const rc=['.zshrc','.bashrc'].map(f=>path.join(os.homedir(),f)).find(f=>fs.existsSync(f));
if(rc){
  let content=fs.readFileSync(rc,'utf8');
  if(content.includes('OPENAI_API_KEY')){
    content=content.replace(/^export OPENAI_API_KEY=.*\n?/m,'');
    fs.writeFileSync(rc,content);
    console.log('Shell: OPENAI_API_KEY removed from '+rc);
  }
}
console.log('All done! Restart terminal to take effect.');
REMOVE_EOF

C Codex CLI 配置

同一个 API Key 也可用于 OpenAI Codex CLI，执行以下命令一键配置： 

node << 'CODEX_EOF'
const fs=require('fs'),path=require('path'),os=require('os');
const apiKey='YOUR_API_KEY';
const dir=path.join(os.homedir(),'.codex');
const sf=path.join(dir,'config.toml');
if(!fs.existsSync(dir))fs.mkdirSync(dir,{recursive:true});
let c=fs.existsSync(sf)?fs.readFileSync(sf,'utf8'):'';
function upsertTop(key,value){
  const line=key+' = '+value;
  const re=new RegExp('^'+key+'\\s*=.*$','m');
  if(re.test(c)){c=c.replace(re,line);return}
  const idx=c.indexOf('[');
  c=idx>=0?line+'\n'+c:line+'\n'+c;
}
upsertTop('model_provider','"local"');
upsertTop('model','"gpt-5.5"');
upsertTop('model_reasoning_effort','"high"');
upsertTop('personality','"pragmatic"');
c=c.replace(/^openai_base_url\s*=.*\n?/m,'');
const providerBlock=[
  '[model_providers.local]',
  'name = "local"',
  'base_url = "{{ service_info.base_url }}/v1"',
  'env_key = "OPENAI_API_KEY"',
  'wire_api = "responses"'
].join('\n');
if(/\[model_providers\.local\][\s\S]*?(?=\n\[|$)/m.test(c)){
  c=c.replace(/\[model_providers\.local\][\s\S]*?(?=\n\[|$)/m,providerBlock);
}else{
  c=c.trimEnd()+'\n\n'+providerBlock;
}
fs.writeFileSync(sf,c.trimEnd()+'\n');
const rc=os.platform()==='win32'?null:['.zshrc','.bashrc'].map(f=>path.join(os.homedir(),f)).find(f=>fs.existsSync(f));
if(apiKey==='YOUR_API_KEY'){
  console.log('Codex config done: '+sf);
  console.log('Replace YOUR_API_KEY with your real API key, or use the generated command after requesting a key.');
}else if(rc){
  const content=fs.readFileSync(rc,'utf8');
  const line='export OPENAI_API_KEY="'+apiKey+'"';
  if(/^export OPENAI_API_KEY=.*/m.test(content)){
    fs.writeFileSync(rc,content.replace(/^export OPENAI_API_KEY=.*/m,line));
    console.log('Updated OPENAI_API_KEY in '+rc);
  }else{
    fs.appendFileSync(rc,'\n'+line+'\n');
    console.log('Added OPENAI_API_KEY to '+rc);
  }
}else{console.log('No .bashrc/.zshrc found. Add manually: export OPENAI_API_KEY='+apiKey)}
console.log('Codex config done: '+sf);
CODEX_EOF

API Key 与 Claude Code 通用，用你申请到的同一个 Key 即可。设置完后运行 codex 即可使用。
